/**
 * current calculator state
 */
export type State = {
  /**
   * next value to be operated on
   * for example 1 + 2 - if user clicks 1 followed by + and then clicks 2 then
   * 2 is the current "next" value
   */
  next: string;
  /**
   * running total result of potentially multiple operations
   */
  total: string;
  /**
   * latest operation
   */
  operation: string;
};

/**
 * action to be dispatched by current button
 */
export type Action = {
  buttonName: string;
};
